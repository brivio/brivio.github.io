杭州驾考网络教学助手

运行环境

chrome+tampermonkey

使用方法

点击进入所要学习的课程，不必点击播放视频，然后插件每隔5秒检测，有弹层出现会有桌面通知，此时应该立刻去处理验证码等弹层，否则计时无效

代码

// ==UserScript==
// @name 杭州驾考网络教学助手
// @namespace 
// @include http://hz.5u5u5u5u.com/studyOnLine*
// @require      http://cdn.bootcss.com/jquery/1.8.3/jquery.min.js
// @run-at       document-end
// @grant        unsafeWindow
// ==/UserScript==

$(function () {
    if (Notification && Notification.permission !== "granted") {
        Notification.requestPermission(function (status) {
            if (Notification.permission !== status) {
                Notification.permission = status;
            }
        });
    }
    var notify_closed = true;

    if (Notification && Notification.permission != "granted") {
        return;
    }

    function show_notify() {
        var options = {
            dir: "ltr",
            lang: "utf-8",
            icon: "http://hz.5u5u5u5u.com/qunle/images/footlogo.png",
            body: "有弹层出现!"
        };
        var n = new Notification("提示", options);
        n.onshow = function () {
            notify_closed = false;
        };
        n.onclick = function () {
        };
        n.onclose = function () {
            notify_closed = true;
        };
        n.onerror = function () {
        };
    }

    setInterval(function () {
        if ($('.xubox_layer').size() > 0) {
            if (notify_closed) {
                show_notify();
            }
        }
        console.log("5秒检查一次……");
    }, 5000);

});
