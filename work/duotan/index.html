客户端处理

  • 链接的scheme从public/settings返回的hybrid_scheme获取，也就是说网页上的链接不一定是app://开头的。
  • app://后的内容先进行base64解码，得到json字符串，然后进行json解析就行了。

ios例子：

//先截获scheme
- (void)webView:(WKWebView *)webView decidePolicyForNavigationAction:(WKNavigationAction *)navigationAction decisionHandler:(void (^)(WKNavigationActionPolicy))decisionHandler {
    decisionHandler(WKNavigationActionPolicyAllow);
    NSURL *url = navigationAction.request.URL;
    if ([[NSString stringWithFormat:@"%@", url] isEqualToString:self.url]) {
        return;
    }
    NSString *scheme = [App shared].settings.hybrid_scheme;

    if ([url.scheme isEqualToString:scheme]) {
        NSUInteger length = [NSString stringWithFormat:@"%@://", scheme].length;
        NSString *data = [url.absoluteString substringWithRange:NSMakeRange(length, url.absoluteString.length - length)];
        [self handleUrl:data];
    }
}
//处理scheme后的内容
- (void)handleUrl:(NSString *)data {
    [super handleUrl:data];
    NSDictionary *dict = data.base64_decode.toDict;
    NSString *method = dict[@"method"];
    NSDictionary *params = dict[@"params"];
    if ([method isEqualToString:@"APPClosePage"]) {
        [self popController];
    }
    else if ([method isEqualToString:@"APPGoTopPage"]) {
        [self popRootController];
    }
    else if ([method isEqualToString:@"APPOpenURL"]) {
        [self pushController:[[AppWebView alloc] initWithTitle:params[@"title"] withUrl:params[@"url"]]];
    }
    else if ([method isEqualToString:@"APPYiDetail"]) {
        DBDetailVC *detailVC = [DBDetailVC new];
        detailVC.yi_id = params[@"yi_id"];
        [self pushController:detailVC];
    }
    else if ([method isEqualToString:@"APPTrendAdd"]) {
        TrendPublicVC *vc = [TrendPublicVC new];
        vc.yi_id = params[@"yi_id"];
        [self pushController:vc];
    }
}
