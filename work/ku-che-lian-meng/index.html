js接口

/*
js接口初始化

初始化完，AppShare等函数才能使用.
 */
AppLoad(fn);

/*
第三方分享

url、img必须都是http形式的链接

例子：
    AppShare(
        '分享标题', 
        '分享内容', 
        'http://www.baidu.com', 
        'http://img1.gtimg.com/v_syjx_pad/pics/hv1/141/89/2077/135079761.jpg'
    )
 */
AppShare(title, content, url, img);

/*
第三方登录

type：微信:wx，QQ:qq，微博:wb
fn:登录成功后的回调，返回值参考如下：
    {
        //登录类型，微信:wx，QQ:qq，微博:wb
        type:"",
        //登录成功后的uid，可以唯一标示特定用户特定登录方式
        uid:"",
        //用户头像
        avatar:"",
        //用户昵称
        user_name:""
    }

例子：
    AppSocialLogin('wx', function (resp) {
        //resp调用方式，获取头像例子：resp.avatar
    });
 */
AppSocialLogin(type,fn);
