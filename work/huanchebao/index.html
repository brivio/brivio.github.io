/**
 *  跳转到出价页面
 *  @params car_id car表id
 *  @params money 价格
 * */
function APPQuote(car_id, money) {
    window.webkit.messageHandlers.APP.postMessage({
        'method': 'APPQuote',
        'car_id': car_id,
        'money': money
    });
}
/**
 * 打开新页面
 * @params title 页面头部标题
 * @params url 页面地址,形式以http://开头
 * */
function APPOpenURL(title, url) {
    window.webkit.messageHandlers.APP.postMessage({
        'method': 'APPOpenURL',
        'url': url,
        'title': title
    });
}
/**
 * 关闭当前页面
 * */
function APPClosePage() {
    window.webkit.messageHandlers.APP.postMessage({
        'method': 'APPClosePage'
    });
}
